function findDeletedNumber(arr, mixArr) {
  if (arr.length === mixArr.length) {
    return 0;
  } else {
    for (let i = 0; i < arr.length; ++i) {
      let found = false;
      for (let j = 0; j < mixArr.length; ++j) {
        if (mixArr[j] === arr[i]) {
          found = true;
          break;
        }
      }
      if (!found) {
        return arr[i];
      }
    }
    return 0;
  }
}
